async function captureAndExtract() { // ... (Your existing camera and canvas setup) ... try { const resp = await fetch(`https://vision.googleapis.com/v1/images:annotate?key=${GOOGLE_API_KEY}`, { method: "POST", body: JSON.stringify({ requests: [{ image: { content: base64 }, features: [{ type: "TEXT_DETECTION" }] }] }) }); const data = await resp.json(); const annotations = data.responses[0].textAnnotations; if (!annotations) throw new Error("No text found"); let mfrPnValue = ""; let anchorY = -1; let anchorXEnd = -1; // 1. HUNT FOR THE ANCHOR // We look at individual word blocks (annotations[1] and up) for (let i = 1; i < annotations.length; i++) { const word = annotations[i].description.toUpperCase(); if (word.includes("MFR") || word.includes("MAN") || word.includes("MFG")) { const v = annotations[i].boundingPoly.vertices; anchorY = v[0].y; // This is our horizontal "lane" anchorXEnd = v[1].x; // This is where the label ends break; } } // 2. GRAB THE NEIGHBOR if (anchorY !== -1) { for (let i = 1; i < annotations.length; i++) { const v = annotations[i].boundingPoly.vertices; const currentY = v[0].y; const currentXStart = v[0].x; const text = annotations[i].description.trim(); // Check: Is it on the same row (+/- 30px) AND to the right of the anchor? if (Math.abs(currentY - anchorY) < 30 && currentXStart > anchorXEnd) { // Ignore colons or distributor "noise" like -ND if (text.length > 4 && !text.includes(":") && !text.toUpperCase().endsWith("-ND")) { mfrPnValue = text; break; // Found it! } } } } // 3. APPLY TO UI document.getElementById("scanMfrPn").value = mfrPnValue.replace(/[\s\(\)\[\]]/g, "").toUpperCase(); } catch(e) { console.error("Extraction error", e); } }
