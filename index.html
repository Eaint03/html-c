async function captureAndScan() { if (processing) return; const video = document.querySelector("#reader video"); if (!video) return; processing = true; if (navigator.vibrate) navigator.vibrate(100); const status = document.getElementById("status"); const canvas = document.getElementById("debugCanvas"); const ctx = canvas.getContext("2d"); status.innerText = "🔍 AI Analyzing Full Label..."; const vW = video.videoWidth; const vH = video.videoHeight; const cropSize = Math.min(vW, vH) * 0.95; canvas.width = 1600; canvas.height = 1600; ctx.drawImage(video, (vW - cropSize)/2, (vH - cropSize)/2, cropSize, cropSize, 0, 0, canvas.width, canvas.height); const base64Image = canvas.toDataURL('image/jpeg', 0.8).split(',')[1]; const GOOGLE_URL = `https://vision.googleapis.com/v1/images:annotate?key=${GOOGLE_API_KEY}`; const requestBody = { requests: [{ image: { content: base64Image }, features: [{ type: "TEXT_DETECTION" }] }] }; try { const response = await fetch(GOOGLE_URL, { method: "POST", body: JSON.stringify(requestBody) }); const data = await response.json(); if (data.responses[0].textAnnotations) { let fullText = data.responses[0].textAnnotations[0].description; let lines = fullText.split('\n'); let pn = ""; let desc = ""; let qty = ""; // --- DEEP ANALYSIS ENGINE --- for (let i = 0; i < lines.length; i++) { let line = lines[i].trim().toUpperCase(); // 1. FIND PART NUMBER (Look for specific patterns) // If it starts with 'BZT' or has many dashes/numbers if (!pn && line.includes('-') && line.length > 8 && !line.includes(' ')) { pn = line; } // 2. FIND DESCRIPTION // Looking for words that describe electronics components if (!desc && (line.includes('DIODE') || line.includes('ZENER') || line.includes('SOD') || line.includes('RESISTOR'))) { desc = line; } // 3. FIND QUANTITY // On DigiKey labels, 100 usually appears near the bottom or after 'Quantity' if (line === "100" || line === "50" || line === "25" || line === "1000") { qty = line; } } // Fallback if logic missed them if (!pn) pn = lines[3] || "Check Label"; if (!desc) desc = lines.find(l => l.length > 15) || "Check Desc"; status.innerText = "🎯 Scan Complete"; // Clean up the Prompt for a professional look const finalInput = prompt( `DIGIKEY DATA DETECTED:\n\n` + `PART #: ${pn}\n` + `DESC: ${desc}\n` + `QTY: ${qty}\n\n` + `Edit below if wrong (Format: PN | DESC | QTY):`, `${pn} | ${desc} | ${qty}` ); if (finalInput) { let parts = finalInput.split('|'); status.innerText = "🚀 Syncing to Sheet..."; await fetch(SCRIPT_URL, { method: "POST", mode: "no-cors", body: JSON.stringify([{ pn: parts[0].trim(), desc: parts[1] ? parts[1].trim() : desc, qty: parts[2] ? parts[2].trim() : qty, mode: mode, date: new Date().toLocaleString() }]) }); status.innerText = "✅ Saved Successfully!"; setTimeout(() => status.innerText = "Ready", 2000); } } } catch (err) { status.innerText = "❌ AI Error. Check connection."; } processing = false; }
