async function captureAndScan() { if (processing) return; const video = document.querySelector("#reader video"); if (!video) return; processing = true; if (navigator.vibrate) navigator.vibrate(100); const status = document.getElementById("status"); const canvas = document.getElementById("debugCanvas"); const ctx = canvas.getContext("2d"); status.innerText = "⏳ AI Reading Label Details..."; // Capture a large square for full context const vW = video.videoWidth; const vH = video.videoHeight; const cropSize = Math.min(vW, vH) * 0.9; const sx = (vW - cropW) / 2; // Note: Ensure cropW is defined or use cropSize const sy = (vH - cropH) / 2; canvas.width = 1500; canvas.height = 1500; ctx.drawImage(video, (vW - cropSize)/2, (vH - cropSize)/2, cropSize, cropSize, 0, 0, canvas.width, canvas.height); const base64Image = canvas.toDataURL('image/jpeg', 0.8).split(',')[1]; const GOOGLE_URL = `https://vision.googleapis.com/v1/images:annotate?key=${GOOGLE_API_KEY}`; const requestBody = { requests: [{ image: { content: base64Image }, features: [{ type: "TEXT_DETECTION" }] }] }; try { const response = await fetch(GOOGLE_URL, { method: "POST", body: JSON.stringify(requestBody) }); const data = await response.json(); if (data.responses[0].textAnnotations) { let fullText = data.responses[0].textAnnotations[0].description; let lines = fullText.split('\n'); // 🧠 AUTO-EXTRACTION LOGIC let pn = ""; let desc = ""; let qty = "1"; for (let i = 0; i < lines.length; i++) { let line = lines[i].trim().toUpperCase(); // Find Part Number (usually has '-' and numbers) if (!pn && line.includes('-') && line.length > 6) { pn = line.replace(/\s/g, ""); } // Find Description (Look for keywords like DIODE, RES, CAP, ZENER) if (!desc && (line.includes('DIODE') || line.includes('ZENER') || line.includes('RES ') || line.includes('CAP '))) { desc = line; } // Find Quantity (Look for the number near "Quantity") if (line.includes('QUANTITY')) { qty = lines[i+1] ? lines[i+1].replace(/\D/g, "") : "1"; } } status.innerText = "🎯 Found: " + (pn || "Unknown"); // Show result to user for confirmation const finalInput = prompt( `Check Data:\n\nPN: ${pn}\nDESC: ${desc}\nQTY: ${qty}\n\n(Edit if needed, use | to separate)`, `${pn} | ${desc} | ${qty}` ); if (finalInput) { let parts = finalInput.split('|'); status.innerText = "🚀 Saving..."; await fetch(SCRIPT_URL, { method: "POST", mode: "no-cors", body: JSON.stringify([{ pn: parts[0].trim(), desc: parts[1] ? parts[1].trim() : desc, qty: parts[2] ? parseInt(parts[2]) : parseInt(qty), mode: mode, date: new Date().toLocaleString() }]) }); status.innerText = "✅ Saved!"; setTimeout(() => status.innerText = "Ready", 2000); } } } catch (err) { status.innerText = "❌ Error reading label."; } processing = false; }
